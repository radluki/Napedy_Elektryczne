\subsection{Wnioski}
\begin{itemize}
	\item Charakterystyka kształtem przypomina hiperbolę, co jest zgodne z modelem matematycznym w którym przyjmuje się, że strumień jest wprost proporcjonalny do prądu, co jest prawdą dla małych wartości, w przeciwnym wypadku następuje nasycenie.
	\item Dodatkowy opór włączony szeregowo do uzwojenia wirnika powoduje obniżenie charakterystyki, tzn. zmniejszenie prędkości obrotowej dla tego samego prądu.
	\item Zmniejszenie napięcia zasilania wirnika powoduje obniżenie charakterystyki, zmniejszenie prędkości jest większe dla mniejszych prądów.
	\item Obserwacje zgadzają się z modelem matematycznym:
	\begin{equation}
		n=\frac{U}{c_2I}-\frac{R_d+R_r}{c_2}
	\end{equation}
	
	\subsection{Wyprowadzenie wzoru nr 1}
	Równania silnika szeregowego, przy założeniu że strumień jest wprost proporcjonalny do prądu:
	\begin{equation}
	U=E+(R_d+R_r)I
	\end{equation}
	\begin{equation}
	E=c\Phi n=c_2 In
	\end{equation}
	
\end{itemize}
